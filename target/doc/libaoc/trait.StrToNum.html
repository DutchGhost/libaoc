pub trait StrToNum<N, S, I> where
    N: FromStr,
    S: AsRef<str>,
    I: Iterator<Item = S>, {
    fn to_num(self) -> Result<Vec<N>, <N as FromStr>::Err>;
}

Takes any Iterator, where the items are implement AsRef. Returns a Vec, where N implements FromStr. Returns an Error if an error occured.

Examples

extern crate libaoc;
use libaoc::{StrToNum};

fn main () {
    let s = "1, 2, 3, 4, 5";
    assert_eq!(vec![1, 2, 3, 4, 5], s.split(", ").to_num().unwrap());

    let s = String::from("1\n2\n3\n4\n5\n6");
    assert_eq!(vec![1, 2, 3, 4, 5, 6], s.lines().to_num().unwrap());
}
